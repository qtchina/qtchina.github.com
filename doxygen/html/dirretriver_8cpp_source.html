00059 void DirRetriver::run()
00060 {
00061     emit enter_remote_dir_retrive_loop();
00062     
00063     int exec_ret_code = -1 ;
00064     command_queue_elem *cmd_elem;
00065     while (this->command_queue.size() > 0) {
00066         cmd_elem = this->command_queue.at(0);
00067         switch(cmd_elem->cmd) {
00068         case SSH2_FXP_READDIR:
00069             this->dir_node_process_queue.insert(std::make_pair(cmd_elem->parent_item, cmd_elem->parent_persistent_index));
00070             exec_ret_code = this->retrive_dir();
00071             break;
00072         case SSH2_FXP_MKDIR:
00073             exec_ret_code = this->mkdir();
00074             break;
00075         case SSH2_FXP_RMDIR:
00076             exec_ret_code = this->rmdir();
00077             break;
00078         case SSH2_FXP_REMOVE:
00079             exec_ret_code = this->rm_file_or_directory_recursively();
00080             break;
00081         case SSH2_FXP_RENAME:
00082             exec_ret_code = this->rename();
00083             break ;
00084         case SSH2_FXP_KEEP_ALIVE:
00085             exec_ret_code = this->keep_alive();
00086             break;
00087         case SSH2_FXP_REALPATH:
00088             exec_ret_code = this->fxp_realpath();
00089             break;
00090         default:    
00091             qDebug()<<tr("unknown command:")<<cmd_elem->cmd;                
00092             break;
00093         }
00094 
00095         // 通知其他人命令执行完成
00096         emit execute_command_finished(cmd_elem->parent_item, cmd_elem->parent_persistent_index,
00097                                       cmd_elem->cmd, exec_ret_code);
00098 
00099         //delet item form queue , stopping infinite cycle
00100         this->command_queue.erase(this->command_queue.begin());
00101         delete cmd_elem;
00102         cmd_elem = 0;
00103         exec_ret_code = -1;
00104     }
00105 
00106     emit this->leave_remote_dir_retrive_loop();
00107 }
