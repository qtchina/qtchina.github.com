00281 static void _q_parseUnixDir(const QStringList &tokens, const QString &userName, QUrlInfo *info)
00282 {
00283     // Unix style, 7 + 1 entries
00284     // -rw-r--r--    1 ftp      ftp      17358091 Aug 10  2004 qt-x11-free-3.3.3.tar.gz
00285     // drwxr-xr-x    3 ftp      ftp          4096 Apr 14  2000 compiled-examples
00286     // lrwxrwxrwx    1 ftp      ftp             9 Oct 29  2005 qtscape -> qtmozilla
00287     if (tokens.size() != 8)
00288         return;
00289 
00290     char first = tokens.at(1).at(0).toLatin1();
00291     if (first == 'd') {
00292         info->setDir(true);
00293         info->setFile(false);
00294         info->setSymLink(false);
00295     } else if (first == '-') {
00296         info->setDir(false);
00297         info->setFile(true);
00298         info->setSymLink(false);
00299     } else if (first == 'l') {
00300         info->setDir(true);
00301         info->setFile(false);
00302         info->setSymLink(true);
00303     }
00304 
00305     // Resolve filename
00306     QString name = tokens.at(7);
00307     if (info->isSymLink()) {
00308         int linkPos = name.indexOf(QLatin1String(" ->"));
00309         if (linkPos != -1)
00310             name.resize(linkPos);
00311     }
00312     info->setName(name);
00313 
00314     // Resolve owner & group
00315     info->setOwner(tokens.at(3));
00316     info->setGroup(tokens.at(4));
00317 
00318     // Resolve size
00319     info->setSize(tokens.at(5).toLongLong());
00320 
00321     QStringList formats;
00322     formats << QLatin1String("MMM dd  yyyy") << QLatin1String("MMM dd hh:mm") << QLatin1String("MMM  d  yyyy")
00323             << QLatin1String("MMM  d hh:mm") << QLatin1String("MMM  d yyyy") << QLatin1String("MMM dd yyyy");
00324 
00325     QString dateString = tokens.at(6);
00326     dateString[0] = dateString[0].toUpper();
00327 
00328     // Resolve the modification date by parsing all possible formats
00329     QDateTime dateTime;
00330     int n = 0;
00331 #ifndef QT_NO_DATESTRING
00332     do {
00333         dateTime = QLocale::c().toDateTime(dateString, formats.at(n++));
00334     }  while (n < formats.size() && (!dateTime.isValid()));
00335 #endif
00336 
00337     if (n == 2 || n == 4) {
00338         // Guess the year.
00339         dateTime.setDate(QDate(QDate::currentDate().year(),
00340                                dateTime.date().month(),
00341                                dateTime.date().day()));

00343     }
00344     if (dateTime.isValid())
00345         info->setLastModified(dateTime);
00346

00348     int permissions = 0;

00350     permissions |= (p[0] == QLatin1Char('r') ? QUrlInfo::ReadOwner : 0);
00351     permissions |= (p[1] == QLatin1Char('w') ? QUrlInfo::WriteOwner : 0);
00352     permissions |= (p[2] == QLatin1Char('x') ? QUrlInfo::ExeOwner : 0);
00353     permissions |= (p[3] == QLatin1Char('r') ? QUrlInfo::ReadGroup : 0);
00354     permissions |= (p[4] == QLatin1Char('w') ? QUrlInfo::WriteGroup : 0);
