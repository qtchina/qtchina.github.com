00052 void SSHDirRetriver::run()
00053 {
00054     emit enter_remote_dir_retrive_loop();
00055     
00056     int exec_ret_code = -1 ;
00057     command_queue_elem *cmd_elem;
00058     while (this->command_queue.size() > 0) {
00059         cmd_elem = this->command_queue.at(0);
00060         switch(cmd_elem->cmd) {
00061         case SSH2_FXP_READDIR:
00062             this->dir_node_process_queue.insert(std::make_pair(cmd_elem->parent_item, cmd_elem->parent_persistent_index));
00063             exec_ret_code = this->retrive_dir();
00064             break;
00065         case SSH2_FXP_MKDIR:
00066             exec_ret_code = this->mkdir();
00067             break;
00068         case SSH2_FXP_RMDIR:
00069             exec_ret_code = this->rmdir();
00070             break;
00071         case SSH2_FXP_REMOVE:
00072             exec_ret_code = this->rm_file_or_directory_recursively();
00073             break;
00074         case SSH2_FXP_RENAME:
00075             exec_ret_code = this->rename();
00076             break ;
00077         case SSH2_FXP_KEEP_ALIVE:
00078             exec_ret_code = this->keep_alive();
00079             break;
00080         case SSH2_FXP_REALPATH:
00081             exec_ret_code = this->fxp_realpath();
00082             break;
00083         default:    
00084             qDebug()<<tr("unknown command:")<<cmd_elem->cmd;                
00085             break;
00086         }
00087 
00088         // 通知其他人命令执行完成
00089         emit execute_command_finished(cmd_elem->parent_item, cmd_elem->parent_persistent_index,
00090                                       cmd_elem->cmd, exec_ret_code);
00091 
00092         //delet item form queue , stopping infinite cycle
00093         this->command_queue.erase(this->command_queue.begin());
00094         delete cmd_elem;
00095         cmd_elem = 0;
00096         exec_ret_code = -1;
00097     }
00098 
00099     emit this->leave_remote_dir_retrive_loop();
00100 }
