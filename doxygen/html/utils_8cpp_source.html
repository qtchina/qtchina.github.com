00024 void strmode ( int mode, char *p )
00025 {
00026     /* print type */
00027     switch (mode & S_IFMT)
00028     {
00029     case S_IFDIR:           /* directory */
00030         *p++ = 'd';
00031         break;
00032     case S_IFCHR:           /* character special */
00033         *p++ = 'c';
00034         break;
00035     case S_IFBLK:           /* block special */
00036         *p++ = 'b';
00037         break;
00038     case S_IFREG:           /* regular */
00039         *p++ = '-';
00040         break;
00041     case S_IFLNK:           /* symbolic link */
00042         *p++ = 'l';
00043         break;
00044         //#ifdef S_IFSOCK
00045     case S_IFSOCK:          /* socket */
00046         *p++ = 's';
00047         break;
00048         //#endif
00049         //#ifdef S_IFIFO
00050     case S_IFIFO:           /* fifo */
00051         *p++ = 'p';
00052         break;
00053         //#endif
00054     default:            /* unknown */
00055         *p++ = '?';
00056         break;
00057     }
00058     /* usr */
00059     if ( mode & S_IRUSR )
00060         *p++ = 'r';
00061     else
00062         *p++ = '-';
00063     if ( mode & S_IWUSR )
00064         *p++ = 'w';
00065     else
00066         *p++ = '-';
00067     switch (mode & (S_IXUSR | S_ISUID))
00068     {
00069     case 0:
00070         *p++ = '-';
00071         break;
00072     case S_IXUSR:
00073         *p++ = 'x';
00074         break;
00075     case S_ISUID:
00076         *p++ = 'S';
00077         break;
00078     case S_IXUSR | S_ISUID:
00079         *p++ = 's';
00080         break;
00081     }
00082     /* group */
00083     if ( mode & S_IRGRP )
00084         *p++ = 'r';
00085     else
00086         *p++ = '-';
00087     if ( mode & S_IWGRP )
00088         *p++ = 'w';
00089     else
00090         *p++ = '-';
00091     switch (mode & (S_IXGRP | S_ISGID))
00092     {
00093     case 0:
00094         *p++ = '-';
00095         break;
00096     case S_IXGRP:
00097         *p++ = 'x';
00098         break;

00100         *p++ = 'S';
00101         break;
00102     case S_IXGRP | S_ISGID:
00103         *p++ = 's';
00104         break;
00105     }
00106     /* other */
00107     if ( mode & S_IROTH )
00108         *p++ = 'r';
00109     else
00110         *p++ = '-';
00111     if ( mode & S_IWOTH )
00112         *p++ = 'w';

00114         *p++ = '-';

00151     char dmode[16] = {0};
00152     int i = 0, v = 0;;
00153     for (i =0 ;i < 4 ; i++) {
00154         v = 0;
00155         if (mode & keys[i*3]) v += 4;
00156         if (mode & keys[i*3+1]) v += 2;
00157         if (mode & keys[i*3+2]) v += 1;
00158         sprintf(dmode+i, "%d", v);
00159         dmode[i+1] = '\0';
00160     }
